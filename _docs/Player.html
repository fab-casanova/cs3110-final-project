Module Player

module Player: sig .. end

The abstract type of values representing players.

type t 
val create_player : string -> Property.t -> t

create_player player_name start is a player with the name player_name and initializes their position at start

val get_name : t -> string

get_name player is the name of player

val get_properties : t -> Property.t list

get_properties player is the list of properties owned by player

val net_worth : t -> int

net_worth player is the net worth of player

val clear_properties : t -> unit

clear_properties player empties the list of properties held by player and sets the owner to the bank

val pp_properties : t -> string

pp_properties player is the string containing the properties of player

val pp_monopolies : t -> string

pp_monopolies player is the string containing the monopolies of player

val player_money : t -> int

player_money player is the amount of money owned by player

val get_position : t -> Property.t

get_position player is the position of player

val change_pos : t -> Property.t -> unit

change_pos player new_pos changes the position of player to new_pos

val add_property : t -> Property.t -> unit

add_property player prop adds prop to the list of properties held by player

val buy_property : t -> Property.t -> unit

buy_property player prop purchases prop for player: adds it to their list of properties, sets player as the owner of prop, decreases their amount of money by the price of prop, and adds prop to the monopoly if it exists

val update_player_money : t -> int -> unit

update_player_money player amount adds amount to player's current amount of money

val calculate_owned_rent : Property.t -> t -> int

calculate_owned_rent prop owner is the rent of prop owned by owner. Value depends on if it is a railroad, utility, or normal colored property

val add_monopoly : t -> Property.space_type -> unit

add_monopoly player prop_type adds prop_type to player's list of monopolies

val has_monopoly : t -> Property.t -> bool

has_monopoly player prop checks if player has a monopoly that includes prop

val check_monopoly : t -> Property.t -> unit

check_monopoly player new_prop adds a new monopoly to player if acquiring new_prop creates a monopoly

val is_building_evenly : Property.t list -> Property.t -> bool -> bool

is_building_evenly props prop is_building takes in list of props and checks if a house can be either built or destroyed on prop depending on the value of is_building. is_building is true if you are building on prop and false if you are destroying. It ensures that you are building evenly over all property types

val roll_dice : unit -> int * int

roll_dice is a tuple of two random numbers from 0-5 and adds 1 to each of them to simulate rolling two die with face values of 1-6

val sum_dice : int * int -> int

sum_dice dice is the sum of the tuple dice which represents two die with values ranging from 1-6

val out_of_cash : int -> t -> bool

out_of_cash amount_owed player is true if amount_owed is greater than player's amount of money and false otherwise

val is_bankrupt : int -> t -> bool

is_bankrupt amount_owed player is true if amount_owed is greater than player's net worth and false otherwise

val can_build_houses_hotel : t -> Property.t -> bool

can_build_houses_hotel player prop is true if player can have houses on prop, if they have a monopoly that includes prop, if they have enough money to buy and a house on prop, if prop is upgradable, and if building on prop ensures that player is building evenly over all property types

val mortgage_allowed : t -> Property.t -> bool

mortgage_allowed player prop is true if player is the owner of prop, prop is not currently being mortgaged, and there are no houses on the color of prop

val no_houses_on_monopoly : t -> Property.t -> bool

no_houses_on_monopoly player prop is true if player doesn't have any houses on the color of prop

val owns_property : t -> Property.t -> bool

owns_property player prop is true is player is an owner of prop

val print_assets : t -> unit

print_assets player prints the assets of player, including player's money, properties, monopolies, and consecutive doubles

val player_status : int -> t -> int

player_status dues player is 0 if player is not out of cash due to their dues, 1 if they are out of cash but not bankrupt, and 2 if they are out of cash and bankrupt

val return_prop_to_bank : t -> Property.t -> unit

return_prop_to_bank player prop removes prop from player's list of properties, sets the owner of prop to the empty string (means it's now owned by the bank), and resets the stage of prop back to Zero

val remove_property : t -> Property.t -> unit

remove_property player prop removes prop from player's list of properties

val swap_owner : t -> t -> Property.t -> unit

swap_owner giver receiver prop swaps ownership of prop from giver to receiver: sets the owner to receiver, adds prop to receiver's list of properties, and removes prop from giver's list of properties

val hand_over_all_properties : t -> t -> unit

hand_over_all_properties giver receiver swaps ownership of all properties owned by giver to receiver: sets the owner to receiver, adds each property to receiver's list of properties, and removes prop from giver's list of properties

val in_jail : t -> bool

in_jail player is true if player is currently in jail

val put_in_jail : t -> unit

put_in_jail player places player in jail for 3 turns

val un_jail : t -> unit

un_jail player removes player from jail

val served_a_turn : t -> unit

served_a_turn player decreases the amount of turns player has to be in jail by 1

val time_left : t -> int

time_left player is the amount of turns left for player to be in jail

val num_doubles : t -> int

num_doubles player is the number of consequtive doubles player has rolled

val reset_doubles : t -> unit

reset_doubles player resets the number of doubles rolled by player to 0

val add_double : t -> unit

add_double player increments the amount of doubles rolled by player

val num_jail_free_cards : t -> int

num_jail_free_cards returns the number of get out of jail free cards player has

val add_jail_free_card : string -> t -> unit

add_jail_free_card gives player a get out of jail free card from the deck represented by jail_card

val owns_jail_free_card : string -> t -> bool

add_jail_free_card returns true if player has a get out of jail free card from the deck jail_card
